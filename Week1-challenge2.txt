#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

void miniMaxSum(vector<int> numbers) {
    // Sort the array to easily pick smallest and largest 4
    sort(numbers.begin(), numbers.end());

    long long minSum = 0;
    long long maxSum = 0;

    // Sum the first 4 numbers for min sum
    for (int i = 0; i < 4; i++) {
        minSum += numbers[i];
    }

    // Sum the last 4 numbers for max sum
    for (int i = 1; i < 5; i++) {
        maxSum += numbers[i];
    }

    // Output both sums in one line
    cout << minSum << " " << maxSum << endl;
}

int main() {
    vector<int> arr(5);
    
    // Reading 5 space-separated integers
    for (int i = 0; i < 5; i++) {
        cin >> arr[i];
    }

    miniMaxSum(arr);

    return 0;
}
